  <script>
    let playing = false;
    let progress = 15;

    function togglePlay() {
      playing = !playing;
      document.getElementById('play-icon').style.display = playing ? 'none' : 'block';
      document.getElementById('pause-icon').style.display = playing ? 'block' : 'none';

      if (playing) {
        animateProgress();
      }
    }

    function animateProgress() {
      if (!playing) return;
      progress += 0.15;
      if (progress > 100) { progress = 0; }
      updateProgress();
      requestAnimationFrame(animateProgress);
    }

    function updateProgress() {
      document.getElementById('progressFill').style.width = progress + '%';
      document.getElementById('progressThumb').style.left = progress + '%';
    }

    function scrub(e) {
      const rect = e.currentTarget.getBoundingClientRect();
      progress = ((e.clientX - rect.left) / rect.width) * 100;
      updateProgress();
    }

    // Toggle hover demo
    const player = document.getElementById('player');
    let demoTimeout;
    player.addEventListener('mouseenter', () => {
      clearTimeout(demoTimeout);
      player.classList.add('demo-show');
    });
    player.addEventListener('mouseleave', () => {
      demoTimeout = setTimeout(() => {
        player.classList.remove('demo-show');
      }, 2000);
    });
  </script>
